429: /*@C
430:    VecGetArrayPair - Accesses a pair of pointers for two vectors that may be common. When not common the first is read only

432:    Logically Collective on Vec

434:    Input Parameter:
435: +  x - the vector
436: -  y - the second vector

438:    Output Parameter:
439: +  xv - location to put pointer to the first array
440: -  yv - location to put pointer to the second array

442:    Level: developer

444: .seealso: VecGetArray(), VecGetArrayRead(), VecRestoreArrayPair()

446: @*/
447: PETSC_STATIC_INLINE PetscErrorCode VecGetArrayPair(Vec x,Vec y,PetscScalar **xv,PetscScalar **yv)
448: {

452:   VecGetArray(y,yv);
453:   if (x != y) {
454:     VecGetArrayRead(x,(const PetscScalar **)xv);
455:   } else {
456:     *xv = *yv;
457:   }
458:   return(0);
459: }

461: /*@C
462:    VecRestoreArrayPair - Returns a pair of pointers for two vectors that may be common. When not common the first is read only

464:    Logically Collective on Vec

466:    Input Parameter:
467: +  x - the vector
468: -  y - the second vector

470:    Output Parameter:
471: +  xv - location to put pointer to the first array
472: -  yv - location to put pointer to the second array

474:    Level: developer

476: .seealso: VecGetArray(), VecGetArrayRead(), VecGetArrayPair()

478: @*/
479: PETSC_STATIC_INLINE PetscErrorCode VecRestoreArrayPair(Vec x,Vec y,PetscScalar **xv,PetscScalar **yv)
480: {

484:   VecRestoreArray(y,yv);
485:   if (x != y) {
