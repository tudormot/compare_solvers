182: /*
183:    Flop counting:  We count each arithmetic operation (e.g., addition, multiplication) separately.

185:    For the complex numbers version, note that
186:        1 complex addition = 2 flops
187:        1 complex multiplication = 6 flops,
188:    where we define 1 flop as that for a double precision scalar.  We roughly approximate
189:    flop counting for complex numbers by multiplying the total flops by 4; this corresponds
190:    to the assumption that we're counting mostly additions and multiplications -- and
191:    roughly the same number of each.  More accurate counting could be done by distinguishing
192:    among the various arithmetic operations.
193:  */

197: #else
198: #define PETSC_FLOPS_PER_OP 1.0

297:      These are used internally in the PETSc routines to keep a count of MPI messages and
298:    their sizes.

300:      This does not work for MPI-Uni because our include/petsc/mpiuni/mpi.h file
301:    uses macros to defined the MPI operations.

303:      It does not work correctly from HP-UX because it processes the
304:    macros in a way that sometimes it double counts, hence

310: /*
311:    Logging of MPI activities
312: */
313: PETSC_STATIC_INLINE PetscErrorCode PetscMPITypeSize(PetscLogDouble *buff,PetscMPIInt count,MPI_Datatype type)
314: {
315:   PetscMPIInt mysize;
316:   PetscErrorCode _myierr;
317:   if (type == MPI_DATATYPE_NULL) return 0;
318:   _myMPI_Type_size(type,&mysize);CHKERRQ(_myierr);
319:   *buff += (PetscLogDouble) (count*mysize);
320:   return 0;
321: }

323: PETSC_STATIC_INLINE PetscErrorCode PetscMPITypeSizeComm(MPI_Comm comm, PetscLogDouble *buff,PetscMPIInt *counts,MPI_Datatype type)
324: {
325:   PetscMPIInt mysize, commsize, p;
326:   PetscErrorCode _myierr;

328:   if (type == MPI_DATATYPE_NULL) return 0;
329:   _myMPI_Comm_size(comm,&commsize);CHKERRQ(_myierr);
330:   _myMPI_Type_size(type,&mysize);CHKERRQ(_myierr);
331:   for (p = 0; p < commsize; ++p) {
332:     *buff += (PetscLogDouble) (counts[p]*mysize);
333:   }
334:   return 0;
335: }

482:   PetscBool      PetscPreLoading = flag;\

484:   PetscLogStage  _stageNum;\
485:   PetscErrorCode _3_ierr; \

487:   PetscPreLoadMax = (int)(PetscPreLoading);\
488:   PetscPreLoadingUsed = PetscPreLoading ? PETSC_TRUE : PetscPreLoadingUsed;\
489:   for (PetscPreLoadIt=0; PetscPreLoadIt<=PetscPreLoadMax; PetscPreLoadIt++) {\
490:     PetscPreLoadingOn = PetscPreLoading;\
491:     _3_PetscBarrier(NULL);CHKERRQ(_3_ierr);\
492:     if (PetscPreLoadIt>0) {\
493:       _3_PetscLogStageGetId(name,&_stageNum);CHKERRQ(_3_ierr);\
494:     } else {\
495:       _3_PetscLogStageRegister(name,&_stageNum);CHKERRQ(_3_ierr); \
496:     }\
497:     _3_PetscLogStageSetActive(_stageNum,(PetscBool)(!PetscPreLoadMax || PetscPreLoadIt));\
498:     _3_PetscLogStagePush(_stageNum);CHKERRQ(_3_ierr);

500: #define PetscPreLoadEnd() \
501:     _3_PetscLogStagePop();CHKERRQ(_3_ierr);\
502:     PetscPreLoading = PETSC_FALSE;\
503:   }\
504: } while (0)
