227: PETSC_STATIC_INLINE PetscErrorCode SNESLogConvergenceHistory(SNES snes,PetscReal res,PetscInt its)
228: {

232:   PetscObjectSAWsTakeAccess((PetscObject)snes);
233:   if (snes->conv_hist && snes->conv_hist_max > snes->conv_hist_len) {
234:     if (snes->conv_hist)     snes->conv_hist[snes->conv_hist_len]     = res;
235:     if (snes->conv_hist_its) snes->conv_hist_its[snes->conv_hist_len] = its;
236:     snes->conv_hist_len++;
237:   }
238:   PetscObjectSAWsGrantAccess((PetscObject)snes);
239:   return(0);
240: }

262: /*
263:     Either generate an error or mark as diverged when a real from a SNES function norm is Nan or Inf
264: */
265: #define SNESCheckFunctionNorm(snes,beta) \
266:   if (PetscIsInfOrNanReal(beta)) {\
267:     if (snes->errorifnotconverged) SETERRQ(PetscObjectComm((PetscObject)snes),PETSC_ERR_NOT_CONVERGED,"SNESSolve has not converged due to Nan or Inf norm");\
268:     else {\
269:       PetscBool domainerror;\
270:       PetscErrorCode MPIU_Allreduce((int*)&snes->domainerror,(int*)&domainerror,1,MPI_INT,MPI_MAX,PetscObjectComm((PetscObject)snes));\
271:       if (domainerror)  snes->reason = SNES_DIVERGED_FUNCTION_DOMAIN;\
272:       else              snes->reason = SNES_DIVERGED_FNORM_NAN;\
273:       return(0);\
274:     }\
275:   }

277: #define SNESCheckKSPSolve(snes)\
278:   {\
279:     KSPConvergedReason kspreason; \
280:     PetscErrorCode KSPGetConvergedReason(snes->ksp,&kspreason);\
281:     if (kspreason < 0) {\
282:       if (kspreason == KSP_DIVERGED_NANORINF) {\
283:         PetscBool domainerror;\
284:         MPIU_Allreduce((int*)&snes->domainerror,(int*)&domainerror,1,MPI_INT,MPI_MAX,PetscObjectComm((PetscObject)snes)); \
285:         if (domainerror)  snes->reason = SNES_DIVERGED_FUNCTION_DOMAIN;\
286:         else              snes->reason = SNES_DIVERGED_LINEAR_SOLVE;                  \
287:         return(0);\
288:       } else {\
289:         if (++snes->numLinearSolveFailures >= snes->maxLinearSolveFailures) {\
290:           PetscInfo2(snes,"iter=%D, number linear solve failures %D greater than current SNES allowed, stopping solve\n",snes->iter,snes->numLinearSolveFailures);\
291:           snes->reason = SNES_DIVERGED_LINEAR_SOLVE;\
292:           return(0);\
293:         }\
294:       }\
295:     }\
296:   }
