179: PETSC_STATIC_INLINE PetscErrorCode TaoLogConvergenceHistory(Tao tao, PetscReal obj, PetscReal resid, PetscReal cnorm, PetscInt totits)
180: {
182:   if (tao->hist_max > tao->hist_len) {
183:     if (tao->hist_obj) tao->hist_obj[tao->hist_len]=obj;
184:     if (tao->hist_resid) tao->hist_resid[tao->hist_len]=resid;
185:     if (tao->hist_cnorm) tao->hist_cnorm[tao->hist_len]=cnorm;
186:     if (tao->hist_lits) {
187:       if (tao->hist_len <= 0) {
188:         tao->hist_lits[0] = totits;
189:       } else {
190:         tao->hist_lits[tao->hist_len]=totits - tao->hist_lits[tao->hist_len-1];
191:       }
192:     }
193:     tao->hist_len++;
194:   }
195:   return(0);
196: }
