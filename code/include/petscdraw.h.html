 69: /*MC

 71:    PetscDrawRealToColor - Maps a real value within an interval to a color.
 72:    The color is an integer value in the range [PETSC_DRAW_BASIC_COLORS to 255]
 73:    that can be passed to various drawing routines.

 75:    Synopsis:
 76:    #include <petscdraw.h>
 77:    int PetscDrawRealToColor(PetscReal value,PetscReal min,PetscReal max)

 79:    Not Collective

 81:    Input Parameter:
 82: +  value - value to map within the interval [min,max]
 83: .  min - lower end of interval
 84: -  max - upper end of interval

 86:    Notes: Values outside the interval [min,max] are clipped.

 88:    Level: intermediate

 90: .seealso: PetscDrawPointPixel(), PetscDrawPoint(), PetscDrawLine(), PetscDrawTriangle(), PetscDrawRectangle()

 92: M*/
 93: PETSC_STATIC_INLINE int PetscDrawRealToColor(PetscReal value,PetscReal min,PetscReal max)
 94: {
 95:   value = PetscClipInterval(value,min,max);
 96:   return PETSC_DRAW_BASIC_COLORS + (int)((255-PETSC_DRAW_BASIC_COLORS)*(value-min)/(max-min));
 97: }
