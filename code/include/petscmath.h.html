667: PETSC_STATIC_INLINE PetscInt PetscPowInt(PetscInt base,PetscInt power)
668: {
669:   PetscInt result = 1;
670:   while (power) {
671:     if (power & 1) result *= base;
672:     power >>= 1;
673:     base *= base;
674:   }
675:   return result;
676: }

678: PETSC_STATIC_INLINE PetscReal PetscPowRealInt(PetscReal base,PetscInt power)
679: {

685:   while (power) {
